function doPost(e) {

  try {

    const spreadsheet =
      SpreadsheetApp.openById(
        "1Nc_XqnrtVChMVIvmuIIIImqTDViOQ10Fy9oVrXpX_2o"
      );

    const action =
      e.parameter.action || "saveOrder";

    //
    // SAVE USER
    //
    if (action === "saveUser") {

      const sheet =
        spreadsheet.getSheetByName(
          "users"
        );

      const data =
        sheet.getDataRange().getValues();

      const email =
        e.parameter.email;

      const existingRow =
        data.findIndex(
          (row, index) =>
            index > 0 &&
            row[2] === email
        );

      // UPDATE EXISTING USER
      if (existingRow !== -1) {

        sheet
          .getRange(
            existingRow + 1,
            1,
            1,
            5
          )
          .setValues([[
            data[existingRow][0],
            e.parameter.name,
            email,
            e.parameter.password,
            e.parameter.role,
          ]]);

      }

      // NEW USER
      else {

        sheet.appendRow([
          Utilities.getUuid(),

          e.parameter.name,

          email,

          e.parameter.password,

          e.parameter.role,
        ]);
      }
    }

    //
    // SAVE ORDER
    //
    else if (
      action === "saveOrder"
    ) {

      const sheet =
        spreadsheet.getSheetByName(
          "orders"
        );

      sheet.appendRow([
        e.parameter.orderId,

        e.parameter.date,

        e.parameter.customerName,

        e.parameter.phone,

        e.parameter.address,

        e.parameter.city,

        e.parameter.pincode,

        e.parameter.products,

        e.parameter.subtotal,

        e.parameter.delivery,

        e.parameter.total,

        "Pending",
      ]);
    }

    //
    // SAVE CART
    //
    else if (
      action === "saveCart"
    ) {

      const sheet =
        spreadsheet.getSheetByName(
          "carts"
        );

      const data =
        sheet.getDataRange().getValues();

      const email =
        e.parameter.email;

      const existingRow =
        data.findIndex(
          (row, index) =>
            index > 0 &&
            row[0] === email
        );

      // UPDATE EXISTING
      if (existingRow !== -1) {

        sheet
          .getRange(
            existingRow + 1,
            2
          )
          .setValue(
            e.parameter.items
          );
      }

      // CREATE NEW
      else {

        sheet.appendRow([
          email,

          e.parameter.items,
        ]);
      }
    }

    //
    // SAVE WISHLIST
    //
    else if (
      action === "saveWishlist"
    ) {

      const sheet =
        spreadsheet.getSheetByName(
          "wishlists"
        );

      const data =
        sheet.getDataRange().getValues();

      const email =
        e.parameter.email;

      const existingRow =
        data.findIndex(
          (row, index) =>
            index > 0 &&
            row[0] === email
        );

      // UPDATE EXISTING
      if (existingRow !== -1) {

        sheet
          .getRange(
            existingRow + 1,
            2
          )
          .setValue(
            e.parameter.items
          );
      }

      // CREATE NEW
      else {

        sheet.appendRow([
          email,

          e.parameter.items,
        ]);
      }
    }

    return ContentService
      .createTextOutput(
        JSON.stringify({
          success: true,
        })
      )
      .setMimeType(
        ContentService.MimeType.JSON
      );

  } catch (error) {

    return ContentService
      .createTextOutput(
        JSON.stringify({
          success: false,
          error: error.toString(),
        })
      )
      .setMimeType(
        ContentService.MimeType.JSON
      );
  }
}
